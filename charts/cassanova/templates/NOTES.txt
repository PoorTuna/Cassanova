{{- else if contains "NodePort" .Values.service.type }}
Get the URL by running:
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "cassanova.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
Get the URL by running:
  kubectl get svc --namespace {{ .Release.Namespace }} {{ include "cassanova.fullname" . }} -w
{{- else }}
Access the UI by running:
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "cassanova.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
  open http://localhost:{{ .Values.service.port }}
{{- end }}
